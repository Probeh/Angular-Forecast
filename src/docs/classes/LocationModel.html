  country           ?: ICountry           ;
  dataSets          ?: string[]           ;

  isAlias           ?: boolean            ;
  key               ?: string             ;
  localizedName     ?: string             ;
  rank              ?: number             ;
  region            ?: ICountry           ;
  timeZone          ?: ITimeZone          ;

}

  public rank              : number             ;

  public timeZone          : ITimeZone          ;

  }
